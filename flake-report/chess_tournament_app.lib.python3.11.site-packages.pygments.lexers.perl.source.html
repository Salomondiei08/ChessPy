207 def analyse_text(text):
208 if shebang_matches(text, r'perl'):
209 return True
210  
211 result = 0
212  
213 if re.search(r'(?:my|our)\s+[$@%(]', text):
214 result += 0.9
215  
216 if ':=' in text:
217 # := is not valid Perl, but it appears in unicon, so we should
218 # become less confident if we think we found Perl with :=
219 result /= 2
220  
221 return result
