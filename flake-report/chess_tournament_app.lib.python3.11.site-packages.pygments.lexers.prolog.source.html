295 def analyse_text(text):
296 if ':- object(' in text:
297 return 1.0
298 elif ':- protocol(' in text:
299 return 1.0
300 elif ':- category(' in text:
301 return 1.0
302 elif re.search(r'^:-\s[a-z]', text, re.M):
303 return 0.9
304 else:
305 return 0.0
