57 class QueryInstance:
58 """
59 A query instance.
60  
61 This is the object on which the actual query operations are performed. The
62 :class:`~tinydb.queries.Query` class acts like a query builder and
63 generates :class:`~tinydb.queries.QueryInstance` objects which will
64 evaluate their query against a given document when called.
65  
66 Query instances can be combined using logical OR and AND and inverted using
67 logical NOT.
68  
69 In order to be usable in a query cache, a query needs to have a stable hash
70 value with the same query always returning the same hash. That way a query
71 instance can be used as a key in a dictionary.
72 """
73  
74 def __init__(self, test: Callable[[Mapping], bool], hashval: Tuple):
75 self._test = test
76 self._hash = hashval
77  
78 def __call__(self, value: Mapping) -> bool:
79 """
80 Evaluate the query to check if it matches a specified value.
81  
82 :param value: The value to check.
83 :return: Wether the value matchs this query.
84 """
85 return self._test(value)

87 def __hash__(self):
88 # We calculate the query hash by using the ``hashval`` object which
89 # describes this query uniquely so we can calculate a stable hash value
90 # by simply hashing it
91 return hash(self._hash)
92  
93 def __repr__(self):
94 return 'QueryImpl{}'.format(self._hash)
95  
96 def __eq__(self, other: object):
97 if isinstance(other, QueryInstance):
98 return self._hash == other._hash
99  
100 return False

102 # --- Query modifiers -----------------------------------------------------
103  
104 def __and__(self, other: 'QueryInstance') -> 'QueryInstance':
105 # We use a frozenset for the hash as the AND operation is commutative
106 # (a & b == b & a) and the frozenset does not consider the order of
107 # elements
108 return QueryInstance(lambda value: self(value) and other(value),
109 ('and', frozenset([self._hash, other._hash])))
110  
111 def __or__(self, other: 'QueryInstance') -> 'QueryInstance':
112 # We use a frozenset for the hash as the OR operation is commutative
113 # (a | b == b | a) and the frozenset does not consider the order of
114 # elements
115 return QueryInstance(lambda value: self(value) or other(value),
116 ('or', frozenset([self._hash, other._hash])))
117  
118 def __invert__(self) -> 'QueryInstance':
119 return QueryInstance(lambda value: not self(value),
120 ('not', self._hash))

123 class Query(QueryInstance):
124 """
125 TinyDB Queries.
126  
127 Allows to build queries for TinyDB databases. There are two main ways of
128 using queries:
129  
130 1) ORM-like usage:
131  
132 >>> User = Query()
133 >>> db.search(User.name == 'John Doe')
134 >>> db.search(User['logged-in'] == True)
135  
136 2) Classical usage:
137  
138 >>> db.search(where('value') == True)
139  
140 Note that ``where(...)`` is a shorthand for ``Query(...)`` allowing for
141 a more fluent syntax.
142  
143 Besides the methods documented here you can combine queries using the
144 binary AND and OR operators:
145  
146 >>> # Binary AND:
147 >>> db.search((where('field1').exists()) & (where('field2') == 5))
148 >>> # Binary OR:
149 >>> db.search((where('field1').exists()) | (where('field2') == 5))
150  
151 Queries are executed by calling the resulting object. They expect to get
152 the document to test as the first argument and return ``True`` or
153 ``False`` depending on whether the documents matches the query or not.
154 """

175 def __getattr__(self, item: str):
176 # Generate a new query object with the new query path
177 # We use type(self) to get the class of the current query in case
178 # someone uses a subclass of ``Query``
179 query = type(self)()
180  
181 # Now we add the accessed item to the query path ...
182 query._path = self._path + (item,)
183  
184 # ... and update the query hash
185 query._hash = ('path', query._path)
186  
187 return query
188  
189 def __getitem__(self, item: str):
190 # A different syntax for ``__getattr__``
191  
192 # We cannot call ``getattr(item)`` here as it would try to resolve
193 # the name as a method name first, only then call our ``__getattr__``
194 # method. By calling ``__getattr__`` directly, we make sure that
195 # calling e.g. ``Query()['test']`` will always generate a query for a
196 # document's ``test`` field instead of returning a reference to the
197 # ``Query.test`` method
198 return self.__getattr__(item)

200 def _generate_test(
201 self,
202 test: Callable[[Any], bool],
203 hashval: Tuple,
204 allow_empty_path: bool = False
205 ) -> QueryInstance:
206 """
207 Generate a query based on a test function that first resolves the query
208 path.
209  
210 :param test: The test the query executes.
211 :param hashval: The hash of the query.
212 :return: A :class:`~tinydb.queries.QueryInstance` object
213 """
214 if not self._path and not allow_empty_path:
215 raise ValueError('Query has no path')
216  
217 def runner(value):
218 try:
219 # Resolve the path
220 for part in self._path:
221 value = value[part]
222 except (KeyError, TypeError):
223 return False
224 else:
225 # Perform the specified test
226 return test(value)
227  
228 return QueryInstance(
229 lambda value: runner(value),
230 hashval
231 )

358 def test(self, func: Callable[[Mapping], bool], *args) -> QueryInstance:
359 """
360 Run a user-defined test function against a dict value.
361  
362 >>> def test_func(val):
363 ... return val == 42
364 ...
365 >>> Query().f1.test(test_func)
366  
367 .. warning::
368  
369 The test fuction provided needs to be deterministic (returning the
370 same value when provided with the same arguments), otherwise this
371 may mess up the query cache that :class:`~tinydb.table.Table`
372 implements.
373  
374 :param func: The function to call, passing the dict as the first
375 argument
376 :param args: Additional arguments to pass to the test function
377 """
378 return self._generate_test(
379 lambda value: func(value, *args),
380 ('test', self._path, func, args)

383 def any(self, cond: Union[QueryInstance, List[Any]]) -> QueryInstance:
384 """
385 Check if a condition is met by any document in a list,
386 where a condition can also be a sequence (e.g. list).
387  
388 >>> Query().f1.any(Query().f2 == 1)
389  
390 Matches::
391  
392 {'f1': [{'f2': 1}, {'f2': 0}]}
393  
394 >>> Query().f1.any([1, 2, 3])
395  
396 Matches::
397  
398 {'f1': [1, 2]}
399 {'f1': [3, 4, 5]}
400  
401 :param cond: Either a query that at least one document has to match or
402 a list of which at least one document has to be contained
403 in the tested document.
404 """
405 if callable(cond):
406 def test(value):
407 return is_sequence(value) and any(cond(e) for e in value)
408  
409 else:
410 def test(value):
411 return is_sequence(value) and any(e in cond for e in value)
412  
413 return self._generate_test(
414 lambda value: test(value),
415 ('any', self._path, freeze(cond))
416 )

464 def fragment(self, document: Mapping) -> QueryInstance:
465 def test(value):
466 for key in document:
467 if key not in value or value[key] != document[key]:
468 return False
469  
470 return True
471  
472 return self._generate_test(
473 lambda value: test(value),
474 ('fragment', freeze(document)),
475 allow_empty_path=True
