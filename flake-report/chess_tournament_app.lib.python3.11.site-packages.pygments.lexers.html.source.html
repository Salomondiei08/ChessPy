269 def get_tokens_unprocessed(self, text):
270 for index, token, value in XmlLexer.get_tokens_unprocessed(self, text):
271 m = re.match('</?xsl:([^>]*)/?>?', value)
272  
273 if token is Name.Tag and m and m.group(1) in self.EXTRA_KEYWORDS:
274 yield index, Keyword, value
275 else:
276 yield index, token, value
