744 class Cycler:
745 """Cycle through values by yield them one at a time, then restarting
746 once the end is reached. Available as ``cycler`` in templates.
747  
748 Similar to ``loop.cycle``, but can be used outside loops or across
749 multiple loops. For example, render a list of folders and files in a
750 list, alternating giving them "odd" and "even" classes.
751  
752 .. code-block:: html+jinja
753  
754 {% set row_class = cycler("odd", "even") %}
755 <ul class="browser">
756 {% for folder in folders %}
757 <li class="folder {{ row_class.next() }}">{{ folder }}
758 {% endfor %}
759 {% for file in files %}
760 <li class="file {{ row_class.next() }}">{{ file }}
761 {% endfor %}
762 </ul>
763  
764 :param items: Each positional argument will be yielded in the order
765 given for each cycle.
766  
767 .. versionadded:: 2.1
768 """
769  
770 def __init__(self, *items: t.Any) -> None:
771 if not items:
772 raise RuntimeError("at least one item has to be provided")
773 self.items = items
774 self.pos = 0
775  
776 def reset(self) -> None:
777 """Resets the current item to the first item."""
778 self.pos = 0
779  
780 @property
781 def current(self) -> t.Any:
782 """Return the current item. Equivalent to the item that will be
783 returned next time :meth:`next` is called.
784 """
785 return self.items[self.pos]
786  
787 def next(self) -> t.Any:
788 """Return the current item, then advance :attr:`current` to the
789 next item.
790 """
791 rv = self.current
792 self.pos = (self.pos + 1) % len(self.items)
793 return rv
794  
795 __next__ = next

812 class Namespace:
813 """A namespace object that can hold arbitrary attributes. It may be
814 initialized from a dictionary or with keyword arguments."""
815  
816 def __init__(*args: t.Any, **kwargs: t.Any) -> None: # noqa: B902
817 self, args = args[0], args[1:]
818 self.__attrs = dict(*args, **kwargs)
819  
820 def __getattribute__(self, name: str) -> t.Any:
821 # __class__ is needed for the awaitable check in async mode
822 if name in {"_Namespace__attrs", "__class__"}:
823 return object.__getattribute__(self, name)
824 try:
825 return self.__attrs[name]
826 except KeyError:
827 raise AttributeError(name) from None
828  
829 def __setitem__(self, name: str, value: t.Any) -> None:
830 self.__attrs[name] = value
831  
832 def __repr__(self) -> str:
833 return f"<Namespace {self.__attrs!r}>"
