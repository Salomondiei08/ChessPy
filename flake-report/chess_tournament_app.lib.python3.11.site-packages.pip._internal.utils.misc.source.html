241 def tabulate(rows: Iterable[Iterable[Any]]) -> Tuple[List[str], List[int]]:
242 """Return a list of formatted rows and a list of column sizes.
243  
244 For example::
245  
246 >>> tabulate([['foobar', 2000], [0xdeadbeef]])
247 (['foobar 2000', '3735928559'], [10, 4])
248 """
249 rows = [tuple(map(str, row)) for row in rows]
250 sizes = [max(map(len, col)) for col in zip_longest(*rows, fillvalue="")]

428 def split_auth_from_netloc(netloc: str) -> NetlocTuple:
429 """
430 Parse out and remove the auth information from a netloc.
431  
432 Returns: (netloc, (username, password)).
433 """
434 if "@" not in netloc:
435 return netloc, (None, None)
436  
437 # Split from the right because that's how urllib.parse.urlsplit()
438 # behaves if more than one @ is present (which can be checked using
439 # the password attribute of urlsplit()'s return value).
440 auth, netloc = netloc.rsplit("@", 1)
441 pw: Optional[str] = None
442 if ":" in auth:
443 # Split from the left because that's how urllib.parse.urlsplit()
444 # behaves if more than one : is present (which again can be checked
445 # using the password attribute of the return value)
446 user, pw = auth.split(":", 1)
447 else:
448 user, pw = auth, None
449  
450 user = urllib.parse.unquote(user)
451 if pw is not None:
452 pw = urllib.parse.unquote(pw)
453  
454 return netloc, (user, pw)

457 def redact_netloc(netloc: str) -> str:
458 """
459 Replace the sensitive data in a netloc with "****", if it exists.
460  
461 For example:
462 - "user:pass@example.com" returns "user:****@example.com"
463 - "accesstoken@example.com" returns "****@example.com"
464 """
465 netloc, (user, password) = split_auth_from_netloc(netloc)
466 if user is None:
467 return netloc

470 password = ""

473 password = ":****"
