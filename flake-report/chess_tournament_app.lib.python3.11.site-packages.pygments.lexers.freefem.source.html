878 def get_tokens_unprocessed(self, text):
879 for index, token, value in CppLexer.get_tokens_unprocessed(self, text):
880 if value in self.operators:
881 yield index, Operator, value
882 elif value in self.types:
883 yield index, Keyword.Type, value
884 elif value in self.fespaces:
885 yield index, Name.Class, value
886 elif value in self.preprocessor:
887 yield index, Comment.Preproc, value
888 elif value in self.keywords:
889 yield index, Keyword.Reserved, value
890 elif value in self.functions:
891 yield index, Name.Function, value
892 elif value in self.parameters:
893 yield index, Keyword.Pseudo, value
894 elif value in self.suppress_highlight:
895 yield index, Name, value
896 else:
897 yield index, token, value
