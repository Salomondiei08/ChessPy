20 class BnfLexer(RegexLexer):
21 """
22 This lexer is for grammar notations which are similar to
23 original BNF.
24  
25 In order to maximize a number of targets of this lexer,
26 let's decide some designs:
27  
28 * We don't distinguish `Terminal Symbol`.
29  
30 * We do assume that `NonTerminal Symbol` are always enclosed
31 with arrow brackets.
32  
33 * We do assume that `NonTerminal Symbol` may include
34 any printable characters except arrow brackets and ASCII 0x20.
  • E501 Line too long (80 > 79 characters)
35 This assumption is for `RBNF <http://www.rfc-base.org/txt/rfc-5511.txt>`_.
36  
37 * We do assume that target notation doesn't support comment.
38  
39 * We don't distinguish any operators and punctuation except
40 `::=`.
41  
42 Though these desision making might cause too minimal highlighting
43 and you might be disappointed, but it is reasonable for us.
44  
45 .. versionadded:: 2.1
46 """
47  
48 name = 'BNF'
49 aliases = ['bnf']
50 filenames = ['*.bnf']
51 mimetypes = ['text/x-bnf']
52  
53 tokens = {
54 'root': [
55 (r'(<)([ -;=?-~]+)(>)',
56 bygroups(Punctuation, Name.Class, Punctuation)),
57  
58 # an only operator
59 (r'::=', Operator),
60  
61 # fallback
62 (r'[^<>:]+', Text), # for performance
63 (r'.', Text),
64 ],
65 }

68 class AbnfLexer(RegexLexer):
69 """
70 Lexer for `IETF 7405 ABNF
71 <http://www.ietf.org/rfc/rfc7405.txt>`_
72 (Updates `5234 <http://www.ietf.org/rfc/rfc5234.txt>`_)
73 grammars.
74  
75 .. versionadded:: 2.1
76 """
77  
78 name = 'ABNF'
79 aliases = ['abnf']
80 filenames = ['*.abnf']
81 mimetypes = ['text/x-abnf']
82  
83 _core_rules = (
84 'ALPHA', 'BIT', 'CHAR', 'CR', 'CRLF', 'CTL', 'DIGIT',
85 'DQUOTE', 'HEXDIG', 'HTAB', 'LF', 'LWSP', 'OCTET',
86 'SP', 'VCHAR', 'WSP')
87  
88 tokens = {
89 'root': [
90 # comment
91 (r';.*$', Comment.Single),
92  
93 # quoted
94 # double quote itself in this state, it is as '%x22'.
95 (r'(%[si])?"[^"]*"', Literal),
96  
97 # binary (but i have never seen...)
98 (r'%b[01]+\-[01]+\b', Literal), # range
99 (r'%b[01]+(\.[01]+)*\b', Literal), # concat
100  
101 # decimal
102 (r'%d[0-9]+\-[0-9]+\b', Literal), # range
103 (r'%d[0-9]+(\.[0-9]+)*\b', Literal), # concat
104  
105 # hexadecimal
106 (r'%x[0-9a-fA-F]+\-[0-9a-fA-F]+\b', Literal), # range
107 (r'%x[0-9a-fA-F]+(\.[0-9a-fA-F]+)*\b', Literal), # concat
108  
109 # repetition (<a>*<b>element) including nRule
110 (r'\b[0-9]+\*[0-9]+', Operator),
111 (r'\b[0-9]+\*', Operator),
112 (r'\b[0-9]+', Operator),
113 (r'\*', Operator),
114  
115 # Strictly speaking, these are not keyword but
116 # are called `Core Rule'.
117 (words(_core_rules, suffix=r'\b'), Keyword),
118  
119 # nonterminals (ALPHA *(ALPHA / DIGIT / "-"))
120 (r'[a-zA-Z][a-zA-Z0-9-]+\b', Name.Class),
121  
122 # operators
123 (r'(=/|=|/)', Operator),
124  
125 # punctuation
126 (r'[\[\]()]', Punctuation),
127  
128 # fallback
129 (r'\s+', Text),
130 (r'.', Text),
131 ],
132 }

215 class PegLexer(RegexLexer):
216 """
217 This lexer is for `Parsing Expression Grammars
218 <https://bford.info/pub/lang/peg.pdf>`_ (PEG).
219  
220 Various implementations of PEG have made different decisions
221 regarding the syntax, so let's try to be accommodating:
222  
223 * `<-`, `←`, `:`, and `=` are all accepted as rule operators.
224  
225 * Both `|` and `/` are choice operators.
226  
227 * `^`, `↑`, and `~` are cut operators.
228  
229 * A single `a-z` character immediately before a string, or
230 multiple `a-z` characters following a string, are part of the
231 string (e.g., `r"..."` or `"..."ilmsuxa`).
232  
233 .. versionadded:: 2.6
234 """
235  
236 name = 'PEG'
237 aliases = ['peg']
238 filenames = ['*.peg']
239 mimetypes = ['text/x-peg']
240  
241 tokens = {
242 'root': [
243 # Comments
244 (r'#.*', Comment.Single),
245  
246 # All operators
247 (r'<-|[←:=/|&!?*+^↑~]', Operator),
248  
249 # Other punctuation
250 (r'[()]', Punctuation),
251  
252 # Keywords
253 (r'\.', Keyword),
254  
255 # Character classes
256 (r'(\[)([^\]]*(?:\\.[^\]\\]*)*)(\])',
257 bygroups(Punctuation, String, Punctuation)),
258  
259 # Single and double quoted strings (with optional modifiers)
260 (r'[a-z]?"[^"\\]*(?:\\.[^"\\]*)*"[a-z]*', String.Double),
261 (r"[a-z]?'[^'\\]*(?:\\.[^'\\]*)*'[a-z]*", String.Single),
262  
263 # Nonterminals are not whitespace, operators, or punctuation
264 (r'[^\s<←:=/|&!?*+\^↑~()\[\]"\'#]+', Name.Class),
265  
266 # Fallback
267 (r'.', Text),
268 ],
269 }
