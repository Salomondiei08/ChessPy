239 def get_lua_functions(version):
240 f = urlopen('http://www.lua.org/manual/%s/' % version)
241 r = re.compile(r'^<A HREF="manual.html#pdf-(?!lua|LUA)([^:]+)">\1</A>')
242 functions = []
243 for line in f:
244 m = r.match(line)
245 if m is not None:
246 functions.append(m.groups()[0])
247 return functions
