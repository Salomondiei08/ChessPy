234 def analyse_text(text):
235 """
236 Check if code contains REBOL header and so it probably not R code
237 """
238 if re.match(r'^\s*REBOL\s*\[', text, re.IGNORECASE):
239 # The code starts with REBOL header
240 return 1.0
241 elif re.search(r'\s*REBOL\s*\[', text, re.IGNORECASE):
242 # The code contains REBOL header but also some text before it
243 return 0.5
