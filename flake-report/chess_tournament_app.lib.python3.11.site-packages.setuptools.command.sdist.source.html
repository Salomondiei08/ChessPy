173 def read_manifest(self):
174 """Read the manifest file (named by 'self.manifest') and use it to
175 fill in 'self.filelist', the list of files to include in the source
176 distribution.
177 """
178 log.info("reading manifest file '%s'", self.manifest)
179 manifest = open(self.manifest, 'rb')
180 for line in manifest:
181 # The manifest must contain UTF-8. See #303.
182 try:
183 line = line.decode('UTF-8')
184 except UnicodeDecodeError:
185 log.warn("%r not UTF-8 decodable -- skipping" % line)
186 continue
187 # ignore comments and blank lines
188 line = line.strip()
189 if line.startswith('#') or not line:
190 continue
191 self.filelist.append(line)
192 manifest.close()
