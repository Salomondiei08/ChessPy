347 class Distribution(_Distribution):
348 """Distribution with support for tests and package data
349  
350 This is an enhanced version of 'distutils.dist.Distribution' that
351 effectively adds the following new optional keyword arguments to 'setup()':
352  
353 'install_requires' -- a string or sequence of strings specifying project
354 versions that the distribution requires when installed, in the format
355 used by 'pkg_resources.require()'. They will be installed
356 automatically when the package is installed. If you wish to use
357 packages that are not available in PyPI, or want to give your users an
358 alternate download location, you can add a 'find_links' option to the
359 '[easy_install]' section of your project's 'setup.cfg' file, and then
360 setuptools will scan the listed web pages for links that satisfy the
361 requirements.
362  
363 'extras_require' -- a dictionary mapping names of optional "extras" to the
364 additional requirement(s) that using those extras incurs. For example,
365 this::
366  
367 extras_require = dict(reST = ["docutils>=0.3", "reSTedit"])
368  
369 indicates that the distribution can optionally provide an extra
370 capability called "reST", but it can only be used if docutils and
371 reSTedit are installed. If the user installs your package using
372 EasyInstall and requests one of your extras, the corresponding
373 additional requirements will be installed if needed.
374  
375 'test_suite' -- the name of a test suite to run for the 'test' command.
376 If the user runs 'python setup.py test', the package will be installed,
377 and the named test suite will be run. The format is the same as
378 would be used on a 'unittest.py' command line. That is, it is the
379 dotted name of an object to import and call to generate a test suite.
380  
381 'package_data' -- a dictionary mapping package names to lists of filenames
382 or globs to use to find data files contained in the named packages.
383 If the dictionary has filenames or globs listed under '""' (the empty
384 string), those names will be searched for in every package, in addition
385 to any names for the specific package. Data files found using these
386 names/globs will be installed along with the package, in the same
387 location as the package. Note that globs are allowed to reference
388 the contents of non-package subdirectories, as long as you use '/' as
389 a path separator. (Globs are automatically converted to
390 platform-specific paths at runtime.)
391  
392 In addition to these new keywords, this class also has several new methods
393 for manipulating the distribution's contents. For example, the 'include()'
394 and 'exclude()' methods can be thought of as in-place add and subtract
395 commands that add or remove packages, modules, extensions, and so on from
396 the distribution.
397 """

811 def include(self, **attrs):
812 """Add items to distribution that are named in keyword arguments
813  
814 For example, 'dist.include(py_modules=["x"])' would add 'x' to
815 the distribution's 'py_modules' attribute, if it was not already
816 there.
817  
818 Currently, this method only supports inclusion for attributes that are
819 lists or tuples. If you need to add support for adding to other
820 attributes in this or a subclass, you can add an '_include_X' method,
821 where 'X' is the name of the attribute. The method will be called with
822 the value passed to 'include()'. So, 'dist.include(foo={"bar":"baz"})'
823 will try to call 'dist._include_foo({"bar":"baz"})', which can then
824 handle whatever special inclusion logic is needed.
825 """
826 for k, v in attrs.items():
827 include = getattr(self, '_include_' + k, None)
828 if include:
829 include(v)
830 else:
831 self._include_misc(k, v)

864 def _exclude_misc(self, name, value):
865 """Handle 'exclude()' for list/tuple attrs without a special handler"""
866 if not isinstance(value, sequence):
867 raise DistutilsSetupError(
868 "%s: setting must be a list or tuple (%r)" % (name, value)
869 )
870 try:
871 old = getattr(self, name)
872 except AttributeError as e:
873 raise DistutilsSetupError(

877 raise DistutilsSetupError(
878 name + ": this setting cannot be changed via include/exclude"
879 )

881 setattr(self, name, [item for item in old if item not in value])

907 """Remove items from distribution that are named in keyword arguments
908  
909 For example, 'dist.exclude(py_modules=["x"])' would remove 'x' from

913  
914 Currently, this method only supports exclusion from attributes that are

916 attributes in this or a subclass, you can add an '_exclude_X' method,
917 where 'X' is the name of the attribute. The method will be called with
918 the value passed to 'exclude()'. So, 'dist.exclude(foo={"bar":"baz"})'
919 will try to call 'dist._exclude_foo({"bar":"baz"})', which can then
920 handle whatever special exclusion logic is needed.
921 """
922 for k, v in attrs.items():
923 exclude = getattr(self, '_exclude_' + k, None)
924 if exclude:
925 exclude(v)
926 else:
927 self._exclude_misc(k, v)

936 def _parse_command_opts(self, parser, args):
937 # Remove --with-X/--without-X options when processing command args
938 self.global_options = self.__class__.global_options
939 self.negative_opt = self.__class__.negative_opt
940  
941 # First, expand any aliases
942 command = args[0]
943 aliases = self.get_option_dict('aliases')
944 while command in aliases:
945 src, alias = aliases[command]
946 del aliases[command] # ensure each alias can expand only once!
947 import shlex
948 args[:1] = shlex.split(alias, True)
949 command = args[0]
950  
951 nargs = _Distribution._parse_command_opts(self, parser, args)
952  
953 # Handle commands that want to consume all remaining arguments
954 cmd_class = self.get_command_class(command)
955 if getattr(cmd_class, 'command_consumes_arguments', None):
956 self.get_option_dict(command)['args'] = ("command line", nargs)
957 if nargs is not None:
958 return []
959  
960 return nargs

962 def get_cmdline_options(self):
963 """Return a '{cmd: {opt:val}}' map of all command-line options
964  
965 Option names are all long, but do not include the leading '--', and
966 contain dashes rather than underscores. If the option doesn't take
967 an argument (e.g. '--quiet'), the 'val' is 'None'.
968  
969 Note that options provided by config files are intentionally excluded.
970 """
971  
972 d = {}
973  
974 for cmd, opts in self.command_options.items():
975  
976 for opt, (src, val) in opts.items():
977  
978 if src != "command line":
979 continue
980  
981 opt = opt.replace('_', '-')
982  
983 if val == 0:
984 cmdobj = self.get_command_obj(cmd)
985 neg_opt = self.negative_opt.copy()
986 neg_opt.update(getattr(cmdobj, 'negative_opt', {}))
987 for neg, pos in neg_opt.items():
988 if pos == opt:
989 opt = neg
990 val = None
991 break
992 else:
993 raise AssertionError("Shouldn't be able to get here")
994  
995 elif val == 1:
996 val = None
997  
998 d.setdefault(cmd, {})[opt] = val
999  
1000 return d

1020 def handle_display_options(self, option_order):
1021 """If there were any non-global "display-only" options
1022 (--help-commands or the metadata display options) on the command
1023 line, display the requested info and return true; else return
1024 false.
1025 """
1026 import sys
1027  
1028 if self.help_commands:
1029 return _Distribution.handle_display_options(self, option_order)
1030  
1031 # Stdout may be StringIO (e.g. in tests)
1032 if not isinstance(sys.stdout, io.TextIOWrapper):
1033 return _Distribution.handle_display_options(self, option_order)
1034  
1035 # Don't wrap stdout if utf-8 is already the encoding. Provides
1036 # workaround for #334.
1037 if sys.stdout.encoding.lower() in ('utf-8', 'utf8'):
1038 return _Distribution.handle_display_options(self, option_order)
1039  
1040 # Print metadata in UTF-8 no matter the platform
1041 encoding = sys.stdout.encoding
1042 errors = sys.stdout.errors
1043 newline = sys.platform != 'win32' and '\n' or None
1044 line_buffering = sys.stdout.line_buffering
1045  
1046 sys.stdout = io.TextIOWrapper(
1047 sys.stdout.detach(), 'utf-8', errors, newline, line_buffering)
1048 try:
1049 return _Distribution.handle_display_options(self, option_order)
1050 finally:
1051 sys.stdout = io.TextIOWrapper(
1052 sys.stdout.detach(), encoding, errors, newline, line_buffering)
