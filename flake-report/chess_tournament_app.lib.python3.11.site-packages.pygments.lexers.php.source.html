112 def get_tokens_unprocessed(self, text):
113 phplexer = PhpLexer(**self.options)
114 curcode = ''
115 insertions = []
116 for match in line_re.finditer(text):
117 line = match.group()
118 if line.startswith('>>> ') or line.startswith('... '):
119 insertions.append((len(curcode),
120 [(0, Generic.Prompt, line[:4])]))
121 curcode += line[4:]
122 elif line.rstrip() == '...':
123 insertions.append((len(curcode),
124 [(0, Generic.Prompt, '...')]))
125 curcode += line[3:]
126 else:
127 if curcode:
128 yield from do_insertions(
129 insertions, phplexer.get_tokens_unprocessed(curcode))
130 curcode = ''
131 insertions = []
132 yield match.start(), Generic.Output, line
133 if curcode:
134 yield from do_insertions(insertions,
135 phplexer.get_tokens_unprocessed(curcode))
