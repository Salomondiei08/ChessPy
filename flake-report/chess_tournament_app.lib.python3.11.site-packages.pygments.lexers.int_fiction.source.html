487 def get_tokens_unprocessed(self, text):
488 # 'in' is either a keyword or an operator.
489 # If the token two tokens after 'in' is ')', 'in' is a keyword:
490 # objectloop(a in b)
491 # Otherwise, it is an operator:
492 # objectloop(a in b && true)
493 objectloop_queue = []
494 objectloop_token_count = -1
495 previous_token = None
496 for index, token, value in RegexLexer.get_tokens_unprocessed(self,
497 text):
498 if previous_token is Name.Variable and value == 'in':
499 objectloop_queue = [[index, token, value]]
500 objectloop_token_count = 2
501 elif objectloop_token_count > 0:
502 if token not in Comment and token not in Text:
503 objectloop_token_count -= 1
504 objectloop_queue.append((index, token, value))
505 else:
506 if objectloop_token_count == 0:
507 if objectloop_queue[-1][2] == ')':
508 objectloop_queue[0][1] = Keyword
509 while objectloop_queue:
510 yield objectloop_queue.pop(0)
511 objectloop_token_count = -1
512 yield index, token, value
513 if token not in Comment and token not in Text:
514 previous_token = token
515 while objectloop_queue:
516 yield objectloop_queue.pop(0)
