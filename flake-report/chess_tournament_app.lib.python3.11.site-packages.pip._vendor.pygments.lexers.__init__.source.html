115 def load_lexer_from_file(filename, lexername="CustomLexer", **options):
116 """Load a lexer from a file.
117  
118 This method expects a file located relative to the current working
119 directory, which contains a Lexer class. By default, it expects the
120 Lexer to be name CustomLexer; you can specify your own class name
121 as the second argument to this function.
122  
123 Users should be very careful with the input, because this method
124 is equivalent to running eval on the input file.
125  
126 Raises ClassNotFound if there are any problems importing the Lexer.
127  
128 .. versionadded:: 2.2
129 """
130 try:
131 # This empty dict will contain the namespace for the exec'd file
132 custom_namespace = {}
133 with open(filename, 'rb') as f:
134 exec(f.read(), custom_namespace)
135 # Retrieve the class `lexername` from that namespace
136 if lexername not in custom_namespace:
137 raise ClassNotFound('no valid %s class found in %s' %
138 (lexername, filename))
139 lexer_class = custom_namespace[lexername]
140 # And finally instantiate it with the options
141 return lexer_class(**options)
142 except OSError as err:
143 raise ClassNotFound('cannot read %s: %s' % (filename, err))
144 except ClassNotFound:
145 raise
146 except Exception as err:
147 raise ClassNotFound('error when loading custom lexer: %s' % err)

150 def find_lexer_class_for_filename(_fn, code=None):
151 """Get a lexer for a filename.
152  
153 If multiple lexers match the filename pattern, use ``analyse_text()`` to
154 figure out which one is more appropriate.
155  
156 Returns None if not found.
157 """
158 matches = []
159 fn = basename(_fn)
160 for modname, name, _, filenames, _ in LEXERS.values():
161 for filename in filenames:
162 if fnmatch(fn, filename):
163 if name not in _lexer_cache:
164 _load_lexers(modname)
165 matches.append((_lexer_cache[name], filename))
166 for cls in find_plugin_lexers():
167 for filename in cls.filenames:
168 if fnmatch(fn, filename):
169 matches.append((cls, filename))
170  
171 if isinstance(code, bytes):
172 # decode it, since all analyse_text functions expect unicode
173 code = guess_decode(code)

175 def get_rating(info):
176 cls, filename = info
177 # explicit patterns get a bonus
178 bonus = '*' not in filename and 0.5 or 0
179 # The class _always_ defines analyse_text because it's included in
180 # the Lexer class. The default implementation returns None which
181 # gets turned into 0.0. Run scripts/detect_missing_analyse_text.py
182 # to find lexers which need it overridden.
183 if code:
184 return cls.analyse_text(code) + bonus, cls.__name__
185 return cls.priority + bonus, cls.__name__
186  
187 if matches:
188 matches.sort(key=get_rating)
189 # print "Possible lexers, after sort:", matches
190 return matches[-1][0]

234 def guess_lexer_for_filename(_fn, _text, **options):
235 """
236 Lookup all lexers that handle those filenames primary (``filenames``)
237 or secondary (``alias_filenames``). Then run a text analysis for those
238 lexers and choose the best result.
239  
240 usage::
241  
242 >>> from pygments.lexers import guess_lexer_for_filename
243 >>> guess_lexer_for_filename('hello.html', '<%= @foo %>')
244 <pygments.lexers.templates.RhtmlLexer object at 0xb7d2f32c>
245 >>> guess_lexer_for_filename('hello.html', '<h1>{{ title|e }}</h1>')
246 <pygments.lexers.templates.HtmlDjangoLexer object at 0xb7d2f2ac>
247 >>> guess_lexer_for_filename('style.css', 'a { color: <?= $link ?> }')
248 <pygments.lexers.templates.CssPhpLexer object at 0xb7ba518c>
249 """
250 fn = basename(_fn)
251 primary = {}
252 matching_lexers = set()
253 for lexer in _iter_lexerclasses():
254 for filename in lexer.filenames:
255 if fnmatch(fn, filename):
256 matching_lexers.add(lexer)
257 primary[lexer] = True
258 for filename in lexer.alias_filenames:
259 if fnmatch(fn, filename):
260 matching_lexers.add(lexer)
261 primary[lexer] = False
262 if not matching_lexers:
263 raise ClassNotFound('no lexer for filename %r found' % fn)
264 if len(matching_lexers) == 1:
265 return matching_lexers.pop()(**options)
266 result = []
267 for lexer in matching_lexers:
268 rv = lexer.analyse_text(_text)
269 if rv == 1.0:
270 return lexer(**options)
271 result.append((rv, lexer))

285 def guess_lexer(_text, **options):
286 """Guess a lexer by strong distinctions in the text (eg, shebang)."""
287  
288 if not isinstance(_text, str):
289 inencoding = options.get('inencoding', options.get('encoding'))
290 if inencoding:
291 _text = _text.decode(inencoding or 'utf8')
292 else:
293 _text, _ = guess_decode(_text)

295 # try to get a vim modeline first
296 ft = get_filetype_from_buffer(_text)

298 if ft is not None:
299 try:
300 return get_lexer_by_name(ft, **options)

304 best_lexer = [0.0, None]
305 for lexer in _iter_lexerclasses():
