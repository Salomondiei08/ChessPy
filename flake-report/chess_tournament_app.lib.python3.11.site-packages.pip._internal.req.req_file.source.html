408 def break_args_options(line: str) -> Tuple[str, str]:
409 """Break up the line into an args and options string. We only want to shlex
410 (and then optparse) the options, not the args. args can contain markers
411 which are corrupted by shlex.
412 """
413 tokens = line.split(" ")
414 args = []
415 options = tokens[:]
416 for token in tokens:
417 if token.startswith("-") or token.startswith("--"):
418 break
419 else:
420 args.append(token)
421 options.pop(0)
422 return " ".join(args), " ".join(options)

453 def join_lines(lines_enum: ReqFileLines) -> ReqFileLines:
454 """Joins a line ending in '\' with the previous line (except when following
455 comments). The joined line takes on the index of the first line.
456 """
457 primary_line_number = None
458 new_line: List[str] = []
459 for line_number, line in lines_enum:
460 if not line.endswith("\\") or COMMENT_RE.match(line):
461 if COMMENT_RE.match(line):
462 # this ensures comments are always matched later
463 line = " " + line
464 if new_line:
465 new_line.append(line)
466 assert primary_line_number is not None
467 yield primary_line_number, "".join(new_line)
468 new_line = []
469 else:
470 yield line_number, line
471 else:
472 if not new_line:
473 primary_line_number = line_number
474 new_line.append(line.strip("\\"))
475  
476 # last line contains \
477 if new_line:
478 assert primary_line_number is not None
479 yield primary_line_number, "".join(new_line)

495 def expand_env_variables(lines_enum: ReqFileLines) -> ReqFileLines:
496 """Replace all environment variables that can be retrieved via `os.getenv`.
497  
498 The only allowed format for environment variables defined in the
499 requirement file is `${MY_VARIABLE_1}` to ensure two things:
500  
501 1. Strings that contain a `$` aren't accidentally (partially) expanded.
502 2. Ensure consistency across platforms for requirement files.
503  
504 These points are the result of a discussion on the `github pull
505 request #3514 <https://github.com/pypa/pip/pull/3514>`_.
506  
507 Valid characters in variable names follow the `POSIX standard
508 <http://pubs.opengroup.org/onlinepubs/9699919799/>`_ and are limited
509 to uppercase letter, digits and the `_` (underscore).
510 """
511 for line_number, line in lines_enum:
512 for env_var, var_name in ENV_VAR_RE.findall(line):
513 value = os.getenv(var_name)
514 if not value:
515 continue
516  
517 line = line.replace(env_var, value)
518  
519 yield line_number, line
