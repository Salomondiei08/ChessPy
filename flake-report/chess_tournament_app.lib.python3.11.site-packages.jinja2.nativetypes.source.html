15 def native_concat(values: t.Iterable[t.Any]) -> t.Optional[t.Any]:
16 """Return a native Python type from the list of compiled nodes. If
17 the result is a single node, its value is returned. Otherwise, the
18 nodes are concatenated as strings. If the result can be parsed with
19 :func:`ast.literal_eval`, the parsed value is returned. Otherwise,
20 the string is returned.
21  
22 :param values: Iterable of outputs to concatenate.
23 """
24 head = list(islice(values, 2))
25  
26 if not head:
27 return None
28  
29 if len(head) == 1:
30 raw = head[0]
31 if not isinstance(raw, str):
32 return raw
33 else:
34 raw = "".join([str(v) for v in chain(head, values)])
35  
36 try:
37 return literal_eval(
38 # In Python 3.10+ ast.literal_eval removes leading spaces/tabs
39 # from the given string. For backwards compatibility we need to
40 # parse the string ourselves without removing leading spaces/tabs.
41 parse(raw, mode="eval")
42 )
43 except (ValueError, SyntaxError, MemoryError):
44 return raw
