218 class SourcesListLexer(RegexLexer):
219 """
220 Lexer that highlights debian sources.list files.
221  
222 .. versionadded:: 0.7
223 """
224  
225 name = 'Debian Sourcelist'
226 aliases = ['debsources', 'sourceslist', 'sources.list']
227 filenames = ['sources.list']
228 mimetype = ['application/x-debian-sourceslist']
229  
230 tokens = {
231 'root': [
232 (r'\s+', Text),
233 (r'#.*?$', Comment),
234 (r'^(deb(?:-src)?)(\s+)',
235 bygroups(Keyword, Text), 'distribution')
236 ],
237 'distribution': [
238 (r'#.*?$', Comment, '#pop'),
239 (r'\$\(ARCH\)', Name.Variable),
240 (r'[^\s$[]+', String),
241 (r'\[', String.Other, 'escaped-distribution'),
242 (r'\$', String),
243 (r'\s+', Text, 'components')
244 ],
245 'escaped-distribution': [
246 (r'\]', String.Other, '#pop'),
247 (r'\$\(ARCH\)', Name.Variable),
248 (r'[^\]$]+', String.Other),
249 (r'\$', String.Other)
250 ],
251 'components': [
252 (r'#.*?$', Comment, '#pop:2'),
253 (r'$', Text, '#pop:2'),
254 (r'\s+', Text),
255 (r'\S+', Keyword.Pseudo),
256 ]
257 }
258  
259 def analyse_text(text):
260 for line in text.splitlines():
261 line = line.strip()
262 if line.startswith('deb ') or line.startswith('deb-src '):
263 return True
