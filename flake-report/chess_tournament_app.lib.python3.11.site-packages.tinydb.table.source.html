42  
43 It provides methods for accessing and manipulating documents.
44  
45 .. admonition:: Query Cache
46  
47 As an optimization, a query cache is implemented using a
48 :class:`~tinydb.utils.LRUCache`. This class mimics the interface of

643 """
644 Read the table data from the underlying storage.
645

673 def _update_table(self, updater: Callable[[Dict[int, Mapping]], None]):
674 """
675 Perform an table update operation.
676  
677 The storage interface used by TinyDB only allows to read/write the
678 complete database data, but not modifying only portions of it. Thus
679 to only update portions of the table data, we first perform a read
680 operation, perform the update on the table data and then write
681 the updated data back to the storage.
682  
683 As a further optimization, we don't convert the documents into the
684 document class, as the table data will *not* be returned to the user.
685 """
686  
687 tables = self._storage.read()
688  
689 if tables is None:
690 # The database is empty
691 tables = {}
692  
693 try:
694 raw_table = tables[self.name]
695 except KeyError:
696 # The table does not exist yet, so it is empty
697 raw_table = {}
698  
699 # Convert the document IDs to the document ID class.
700 # This is required as the rest of TinyDB expects the document IDs
701 # to be an instance of ``self.document_id_class`` but the storage
702 # might convert dict keys to strings.
703 table = {
704 self.document_id_class(doc_id): doc
705 for doc_id, doc in raw_table.items()
706 }
707  
708 # Perform the table update operation
709 updater(table)
710  
711 # Convert the document IDs back to strings.
712 # This is required as some storages (most notably the JSON file format)
713 # don't support IDs other than strings.
714 tables[self.name] = {
715 str(doc_id): doc
716 for doc_id, doc in table.items()
717 }
718  
719 # Write the newly updated data back to the storage
720 self._storage.write(tables)
721  
722 # Clear the query cache, as the table contents have changed
723 self.clear_cache()
